  const chain = {

    username: 'user',
    password: 'pwd',
    udp: true,
    'dialer-proxy': '🇯🇵 日本中转'
  };
  config.proxies = config.proxies || [];
  config.proxies.push(chain);

  const g = config['proxy-groups']?.[0];

  return config;
}
